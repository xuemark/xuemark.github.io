3. Create EKS Cluster and Nodegroup

This is link for return home page.

1. Environment

export AWS_REGION=ap-southeast-1
export CLUSTER_NAME=eks-test
export NODEGROUP=$CLUSTER_NAME-nodegroup
export TOMCAT_VERSION=9.0.37

2. Create EKS Cluster and Nodegroup

Create Cluster Script

cat <<EOF > create_cluster.sh
eksctl create cluster \
--name $CLUSTER_NAME \
--version 1.17 \
--region $AWS_REGION \
--vpc-public-subnets xxx \
--vpc-private-subnets yyy \
--nodegroup-name $CLUSTER_NAME-nodegroup \
--node-type t3.small \
--node-volume-size 5 \
--nodes 1 \
--nodes-min 1 \
--nodes-max 5 \
--node-private-networking \
--alb-ingress-access \
--managed \
--asg-access \
--full-ecr-access
EOF

Get and replace subnet id

subnets=$(aws ec2 describe-subnets  --filters Name=tag:Name,Values="Public Subnet 1","Public Subnet 2","Public Subnet 3" --region $AWS_REGION --query "Subnets[*].{SubnetId:SubnetId}" --o text)

subnets=${subnets//[[:space:]]/,}

sed -i "s/xxx/${subnets}/" create_cluster.sh

subnets=$(aws ec2 describe-subnets  --filters Name=tag:Name,Values="Private Subnet 1","Private Subnet 2","Private Subnet 3" --region $AWS_REGION --query "Subnets[*].{SubnetId:SubnetId}" --o text)

subnets=${subnets//[[:space:]]/,}

sed -i "s/yyy/${subnets}/" create_cluster.sh

cat create_cluster.sh

Run create_cluster.sh

sh create_cluster.sh

wait about 10-15 minutes
